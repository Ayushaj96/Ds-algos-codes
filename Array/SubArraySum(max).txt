#include <iostream>
using namespace std;
int main() {
    int n,maxsum=0,cursum,l=-1,r=-1;
    cin>>n;

    int a[1000],b[1000];

    cin>>a[0];
    b[0]=a[0];

    for(int i=1;i<n;i++)
    {
        cin>>a[i];
        b[i]=b[i-1]+a[i];
    }

for(int i=0;i<n;i++)
{
    for(int j=i;j<n;j++)
    {
        cursum=0;
        cursum=b[j]-b[i-1];
        cout<<cursum<<" ";
        if(cursum>maxsum)
        {
            maxsum=cursum;
            l=i;
            r=j;
        }


    }
    cout<<endl;
}

cout<<"max sum is: "<<maxsum<<endl;
cout<<"sub array is: ";
for(int i=l;i<=r;i++)
    cout<<a[i]<<" ";

}
