#include <iostream>
#include<cstring>
using namespace std;

class Car{
private:
    int price;
public:
    int model_no;
    char *name;
    
    //Constructor 
    Car(){
        //Override the default Constructor
        name = NULL;
        cout<<"Making a car.."<<endl;
    }
    Car(int p,int mn,char *n){
        price = p;
        model_no = mn;
        int l  = strlen(n);
        name = new char[l+1];
        strcpy(name,n);
        
    }
   
   //Deep Copy Constructor 
   Car(Car &X){
        price = X.price;
        model_no = X.model_no;
        int l = strlen(X.name);
        name = new char[l+1];
        strcpy(name,X.name);
    }

    }

};
