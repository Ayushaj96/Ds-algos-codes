#include <iostream>
using namespace std;
int main() {
    char a[10][100];
    
    int n;
    cin>>n;
    
    cin.ignore();    //to ignore the enter after n
    
    for(int i=0;i<n;i++){
        cin.getline(a[i],100); //Each row of the character will get filled
    }
    
    //print
    for(int i=0;i<n;i++){
        cout<<a[i]<<endl;
    }
}
